\begin{frame}{Referrals, learning, and efficiency --}
\begin{itemize}

\item Physician practice styles / supply-side factors are a large source of inefficient variations in health care

% variations *within* areas

\item PCPs have substantial influence over choice of specialists
\medskip
\item How well do PCPs learn about specialist quality and direct patients toward better specialists?

\item What is the scope to improve outcomes and reduce costs via better referrals?

\bigskip \pause

\item Related literature:
\bit
    \item practice variations (esp.~\emph{within} geographic areas)
    \item patient-sharing networks, referral networks
%    \item referral networks -- care fragmentation, career implications
%   Johnson, Sarsons papers
    \item physician learning -- structural models
    \item consumer response to quality
    \item physician agency -- informational frictions, subj.~preferences
\eit

\end{itemize}
\end{frame}

\begin{frame}{Structural learning model}
\begin{itemize}

% How we are thinking about this...

\item Bayesian learning, multi-armed bandit framework
\smallskip

\item Uncertainty about quality of each specialist: \\ $p_j$ -- true probability of success or failure with each patient
\medskip \pause 

\item PCP $i$ utility from sending patient $t$ to specialist $j$ (at date $t$): \\
\smallskip

$U_{ijt} \equiv \alpha m_{ijt} + f(e_{ijt}) + u(x_{ijt}) + \xi_j + \epsilon_{ijt}$
\smallskip

\bit
    \item $m_{ijt}$ -- mean of beliefs about $p_j$ \\
    (based on history of patient outcomes before $t$)
    \item $e_{ijt}$ -- experience with specialist $j$ \\
    (number or proportion of past patients sent to $j$)
    \item $x_{ijt}$ -- patient-specific factors (e.g., distance)
\eit 
\smallskip

\item Model quantifies learning and frictions:
\bit
    \item Speed of learning vs.~stickiness of beliefs
    \item PCP's subjective value of professional relationships 
    \item Experimentation with unfamiliar specialists
\eit

\end{itemize}
\end{frame}

%%%

\begin{frame}[t]{Empirical specification}
\vskip-12pt

\[
    \Pr \left( j  \, | \, \dots \right) = 
    \frac{\exp \left( \pi_1 m_{ijt} + \pi_2 m_{ijt} e_{ijt} + \pi_3 e_{ijt} + \pi_4 x_{ijt} + \xi_j \right)}{\sum_{k \in J_{it}} \exp \left( \pi_1 m_{ikt} + \pi_2 m_{ikt} e_{ikt} + \pi_3 e_{ikt} + \pi_4 x_{ikt} + \xi_k \right)}
\]

\begin{itemize}

\item Multinomial logit with specialist fixed effects
\medskip

\item Variable definitions:
\bit
    \item $m_{ijt}$ -- failure rate among patients sent to $j$ over past five years
    \item $e_{ijt}$ -- proportion of patients sent to $j$ over past five years
    \item $x_{ijt}$ -- dist.~from patient to hospital where $j$ primarily operates
\eit
\smallskip

\item Choice set ($J_{it}$): all specialists in HRR who operate in same year and are w/in 150 miles of patient 
\medskip

\item Identification comes from differences \emph{across} PCPs in outcomes of patients sent to \emph{same} specialists
\medskip

\item Estimated separately in each HRR

\end{itemize} 
\end{frame}

\begin{frame}{Conclusions}
\begin{itemize}

\item Using variation in patient outcomes across PCPs referring patients to the same specialists, we estimate a small but significant responsiveness of referrals to outcomes
\bigskip

\item Substantial improvements are possible if referrals can be reallocated \emph{within} geographic areas
\bigskip

\item At the market level, responsiveness to outcomes is
\bit
    \item Positively related to the number of physicians
    \item Negatively related to market concentration among specialists
\eit

\end{itemize}
\end{frame}
